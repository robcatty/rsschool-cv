function countBy(x, n) {
  let z = [];

  for (let i = 1; z.length < n; i++) {
    if (i % x === 0) {
      z.push(i);
    }
  }

  return z;
}
```

function countSheeps(arrayOfSheep) {
  let count = 0;
  for (let i = 0; i < arrayOfSheep.length; i++) {
    if (arrayOfSheep[i] === true) {
      count++;
    }
  }
  return count;
}
```
